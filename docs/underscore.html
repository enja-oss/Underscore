    iterator = iterator == null ? _.identity : lookupIterator(iterator);
    var value = iterator.call(context, obj);
    var low = 0, high = array.length;
    while (low < high) {

      iterator.call(context, array[mid]) < value ? low = mid + 1 : high = mid;

        return array[i] === item ? i : -1;
